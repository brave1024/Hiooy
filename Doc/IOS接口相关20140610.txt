快速购买下单
http://linux.hiooy.com/ecstore/index.php/wap/order-create.html
    1.参数名：member_id {
        描述：会员id
        类型：varchar
        必填：是
    }
    2.参数名：addr_id {
        描述：收货地址id
        类型：varchar
        必填：是
    }
    3.参数名：seller[ID][shipping_id]{
        描述：对应商户配送方式id，其中ID为对应商户的id号，
        类型：varchar
        必填：是
    }
    4.参数名：seller[ID][goods_id]{
        描述：商品id
        类型：varchar
        必填：是
    }
    5.参数名：seller[ID][product_id]{
        描述：货品id，
        类型：varchar
        必填：是
    }
    6.参数名：seller[ID][num]{
        描述：数量
        类型：varchar
        必填：是
    }
    7.参数名：payment_app_id {
        描述：支付方式id
        类型：varchar
        必填：是
    }
    8.参数名：memo {
        描述：客户留言
        类型：longtext
        必填：是
    }
    9.参数名：is_fast{
        描述：快速购买标示
        类型：varchar
        必填：是，这里为1
    }
